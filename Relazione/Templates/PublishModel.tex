    \begin{tabularx}{\textwidth}{|l|>{\raggedright\arraybackslash}X|}
        \hline
        \textbf{Field} & \textbf{Content} \\
        \hline
        \textbf{Name} & Publish Model \\
        \hline
        \textbf{Level} & User Goal \\
        \hline
        \textbf{Description} & The Model Provider publishes a model on the platform. \\
        \hline
        \textbf{Pre-conditions} & The Model Provider is logged in. \\
        \hline
        \textbf{Post-conditions} & The model is uploaded in PENDING\_REVIEW state. \\
        \hline
        \textbf{Actors} & Model Provider\\
        \hline
        \textbf{Main Flow} & 
        \normallist{
            \item The model provider accesses the "Publish Model" page.
            \item The model provider enters the model name in the "Model Name" text box.
            \item The model provider enters the model description in the "Description" text box.
            \item The model provider enters the provider cost per token (how much he wants to receive for each token) in the "Cost per Token Provider (€)" text box.
            \item The model provider presses "BROWSE FILES".
            \item The system opens the file manager for selecting ".safetensors" and ".json" files.
            \item The model provider selects the files of the model and confirms.
            \item The system closes the file manager.
            \item The model provider presses "PUBLISH".
            \item The system shows the message "The model is waiting to be reviewed, it will be published after it has been validated".
        }
        \\
        \hline
        \textbf{Alternative Flows} & 
        \altblock{7.}{Incomplete selection:}{
            \item If the model provider doesn't select at least a ".safetensors" or a ".json" file, the system shows an error message.
            \item Return to step 5.
        }
        \\
        \hline
    \end{tabularx}
