	\begin{tabularx}{\textwidth}{|l|>{\raggedright\arraybackslash}X|}
		\hline
		\textbf{Field} & \textbf{Content} \\
		\hline
		\textbf{Name} & Verify Model \\
		\hline
		\textbf{Level} & User Goal \\
		\hline
		\textbf{Description} & The supervisor performs tests regarding the ethics and performance of a model waiting for review, to determine if it can be published on the site and its hosting cost. \\
		\hline
		\textbf{Actors} & Supervisor \\
		\hline
		\textbf{Unit Test} & \texttt{VerificationServiceTest} \ref{lst:verification_service_test} \\
		\hline
		\textbf{Sequence Diagram} & Figure \ref{fig:seq_verify_model} (see Section \ref{subsec:VerificationService}) \\
		\hline
		\textbf{Pre-conditions} & There is at least one model in \texttt{PENDING\_REVIEW} state. \\
		\hline
		\textbf{Post-conditions} & The reviewed model is in \texttt{APPROVED} or \texttt{REJECTED} state. \\
		\hline
		\textbf{Main Flow} & 
		\normallist{
			\item The supervisor accesses the \texttt{Verification Dashboard} page.
			\item The supervisor selects a model.
			\item The supervisor presses \texttt{RUN AUTOMATED TESTS}.
			\item The system loads the model on a free GPU.
			\item The system runs benchmarks and ethics tests, showing the output results.
			\item (Optional) The supervisor enters a prompt in the \texttt{Manual Ethics Test} text box and presses \texttt{TEST} to manually test its behavior.
			\item The supervisor selects one of the following options:
			\begin{itemize}
				\item Approve Model:
				\begin{enumerate}[label=\arabic*a., start=8, nosep]
					\item The supervisor enters the hosting cost per token in the \texttt{Platform Cost per Token (€)} text box.
					\item The supervisor presses \texttt{APPROVE MODEL}.
				\end{enumerate}
				
				\item Reject Model:
				\begin{enumerate}[label=\arabic*b., start=8, nosep]
					\item The supervisor presses \texttt{REJECT MODEL}.
					\item The system shows a message asking for a rejection reason.
					\item The supervisor enters the reason and presses \texttt{CONFIRM}.
				\end{enumerate}
			\end{itemize}
			\item The system frees the used GPU.
		}   
		\\
		\hline
		\textbf{Alternative Flows} & 
		\altblock{4.}{No free GPU:}{
			\item The system finds that there are no free GPUs.
			\item The message \textit{"No GPU available, try again later"} is shown to the supervisor.
		}
		\\
		\hline
	\end{tabularx}
\end{table}
