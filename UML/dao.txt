package dao.interfaces #FFF8DC {

    interface "GenericDao<T>" as GenericDao {
        + save(entity: T): void
        + findById(id: int): T
        + findAll(): List<T>
        + update(entity: T): void
        + delete(id: int): void
    }

    ' Raggruppo le interfacce per risparmiare larghezza
    together {
        interface AiModelDao {
            + findByStatus(status: ModelStatus): List<AiModel>
        }

            + findByStatus(status: ComplaintStatus): List<Complaint>
        }
    }

    together {
        interface GpuDao {

    }
}
